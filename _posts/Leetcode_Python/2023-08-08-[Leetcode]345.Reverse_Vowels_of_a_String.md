## My code


```python
class Solution:
    def reverseVowels(self,s: str) -> str:
        vowels = {'a','e','i','o','u'}
        li = 0
        ri = len(s) - 1
        s = list(s)
        while li < ri:
            if s[li].lower() in vowels and s[ri].lower() in vowels:
                s[li],s[ri] = s[ri],s[li]
                li += 1
                ri -= 1
            elif s[ri].lower() not in vowels:
                ri -= 1
            elif s[li].lower() not in vowels:
                li += 1
        
        return ''.join(s)
```
