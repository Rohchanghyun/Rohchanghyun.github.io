## My code


```python
class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:
        start = end = 0
        max_count = count = 0

        while end < len(nums):
            if nums[end] == 1:
                end += 1
                count += 1
            elif nums[end] == 0 and k > 0:
                end += 1
                k -= 1
                count += 1
            else:
                while nums[start] == 1:
                    start += 1
                    count -= 1
                start += 1
                end += 1

            max_count = max(max_count,count)

        return max_count
        
```
